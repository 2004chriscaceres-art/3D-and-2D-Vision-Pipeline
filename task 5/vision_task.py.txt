import cv2
from matplotlib import pyplot as plt

# 1. Load the image
img = cv2.imread('cup.png', cv2.IMREAD_GRAYSCALE)

# 2. Preprocessing: Blur the image to reduce grain/noise
# A larger kernel (e.g., 7x7) makes it blurrier
blurred = cv2.GaussianBlur(img, (7, 7), 0)

# 3. Edge Detection: Find the outlines
# The numbers (50, 150) are your thresholds. 
# Try changing them to see how the edges change!
edges = cv2.Canny(blurred, 50, 150)

# 4. Display the result
plt.figure(figsize=(10, 5))
plt.subplot(1, 2, 1)
plt.imshow(img, cmap='gray')
plt.title('Original Image')

plt.subplot(1, 2, 2)
plt.imshow(edges, cmap='gray')
plt.title('Detected Edges')
plt.show()
